USAGE EXAMPLE:
--------------

// Initialize
LevelingSystem leveling;
ComboSystem combos;
leveling.initialize();
combos.initialize(&leveling);

// Per frame
leveling.update(deltaTime);
combos.update(deltaTime);

// On enemy kill
leveling.addXP(EnemyXPRewards::LARGE_DOG);

// On player attack
leveling.addWeaponXPFromDamage("sword", damageDealt);

// Check for combo
auto* combo = combos.checkForCombo(currentWeapon);
if (combo) {
    combos.startCombo(*combo);
    float damage = baseDamage * combo->damageMultiplier;
}

// Use abilities
if (leveling.hasAbility("regeneration")) {
    leveling.applyRegeneration(dt, health, maxHealth);
}

if (leveling.canRevive()) {
    leveling.useRevive(health, maxHealth);
}
